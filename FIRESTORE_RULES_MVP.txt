// Paste into Firebase Console > Firestore Database > Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{uid} {
      allow read, write: if request.auth != null && request.auth.uid == uid;
    }

    function isTripMember(tripId) {
      return request.auth != null &&
        exists(/databases/$(database)/documents/trips/$(tripId)/members/$(request.auth.uid));
    }

    match /trips/{tripId} {
      // Allow signed-in users to discover trips by invite code.
      allow read: if request.auth != null;
      // Allow creating a new trip if creator is the signed-in user.
      allow create: if request.auth != null &&
        request.resource.data.createdBy == request.auth.uid;
      // Only members can update/delete existing trip docs.
      allow update, delete: if isTripMember(tripId);

      match /members/{uid} {
        // Allow users to create/update their own membership doc (join flow).
        allow create, update: if request.auth != null && request.auth.uid == uid;
        allow read: if isTripMember(tripId);
        allow delete: if isTripMember(tripId);
      }

      match /expenses/{expenseId} {
        allow read, write: if isTripMember(tripId);

        match /splits/{uid} {
          allow read, write: if isTripMember(tripId);
        }
      }

      match /itinerary/{dayId} {
        allow read, write: if isTripMember(tripId);

        match /items/{itemId} {
          allow read, write: if isTripMember(tripId);
        }
      }
    }
  }
}
